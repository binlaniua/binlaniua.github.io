docker ) mysql

2019年7月26日
| docker 
expost=端口号 \
logPath=日志目录 \
dataPath=数据目录 \
rootPass=root密码 \
docker \
    run \
    -p $expose:3306 \
    -v $logPath:/logs \
    -v $dataPath:/var/lib/mysql \
    -e MYSQL_ROOT_PASSWORD=$rootPass \
    --name mysql \
    -d mysql:5.7
